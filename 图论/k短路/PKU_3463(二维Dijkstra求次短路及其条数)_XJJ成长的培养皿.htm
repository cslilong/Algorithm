PKU_3463(二维Dijkstra求次短路及其条数)
2009-08-11 15:27
PKU 3463:求最短路和次短路的条数之和，问了杜牛之后得知和必须改用二维的Dijkstra来做，这道题必须深刻地理解Dijkstra的定标技术，而这也是
Dijkstra 为什么只能处理正权边的图，这是因为Dijkstra中的一个点的最短路一旦被定标，即从队列中pop出来之后就不可能再出现更优的值了，而一旦出现负权边就会破坏这条性质。这道题其实也是一样的，一个点的最短路和次短路一旦被定标，也不可能出现更优的值了。所以这里要把数组都搞成2维的，dist[] [0]记录最短路，dist[][1]记录次短路，再搞一个cnt数组进行计数，而队列的操作和Dijkstra几乎是一样的，只不过队列中的每个节点多了一个标志，标志该节点是最短路还是次短路，每次pop出来后对相连的点的内容及队列进行更新。具体的更新细节见代码，应该很容易理解。


#include <stdio.h>
#include <memory.h>
#include <vector>
#include <queue>
using namespace std;

struct Node
{
int vex, weight, flg;
}node;
int n, m, s, t, dist[1005][2], cnt[1005][2];
bool visited[1005][2];
vector< Node > map[1005];
priority_queue< Node > q;

bool operator < ( Node a, Node b )
{
return a.weight > b.weight;
}

int dijkstra()
{
int i;
while( !q.empty() ) q.pop();
for(i=0; i<n; ++i)
{
dist[i][0] = dist[i][1] = 0x7fffffff;
cnt[i][0] = cnt[i][1] = 0;
visited[i][0] = visited[i][1] = false;
}
dist[s][0] = 0;
cnt[s][0] = 1;
node.flg = 0;
node.vex = s;
node.weight = 0;
q.push(node);
while ( !q.empty() )
{
node = q.top();
q.pop();
if( visited[ node.vex ][ node.flg ] ) continue;
visited[ node.vex ][ node.flg ] = true;
int size = map[ node.vex ].size(), i;
for(i=0; i<size; ++i)
{
int k = map[ node.vex ][i].vex;
int tmp = map[ node.vex ][i].weight + node.weight;
Node store;
if( tmp < dist[k][0] )
{
if( dist[k][0] != 0x7fffffff )
{
dist[k][1] = dist[k][0];
cnt[k][1] = cnt[k][0];
store.flg = 1;
store.vex = k;
store.weight = dist[k][1];
q.push(store);
}
dist[k][0] = tmp;
cnt[k][0] = cnt[ node.vex ][ node.flg ];
store.flg = 0;
store.vex = k;
store.weight = dist[k][0];
q.push(store);
}
else if( tmp == dist[k][0] )
{
cnt[k][0] += cnt[ node.vex ][ node.flg ];
}
else if( tmp < dist[k][1] )
{
dist[k][1] = tmp;
cnt[k][1] = cnt[ node.vex ][ node.flg ];
store.flg = 1;
store.vex = k;
store.weight = dist[k][1];
q.push(store);
}
else if( tmp == dist[k][1] )
{
cnt[k][1] += cnt[ node.vex ][ node.flg ];
}
}
}
int ret = cnt[t][0];
if( dist[t][0] == dist[t][1] - 1 ) ret += cnt[t][1];
return ret;
}

int main()
{
int test;
scanf("%d", &test);
while ( test-- )
{
scanf("%d %d", &n, &m);
int i;
for(i=0; i<n; ++i)
{
if( !map[i].empty() ) map[i].clear();
}
while ( m-- )
{
int a;
scanf("%d %d %d", &a, &node.vex, &node.weight);
a--;
node.vex--;
node.flg = 0;
map[a].push_back(node);
}
scanf("%d %d", &s, &t);
s--;
t--;
printf("%d\n", dijkstra());
}
return 0;
}
